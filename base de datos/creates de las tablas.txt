DROP TABLE IF EXISTS likeUsuario;
DROP TABLE IF EXISTS guardado;
DROP TABLE IF EXISTS respuestas;
DROP TABLE IF EXISTS posts;
DROP TABLE IF EXISTS temas;
DROP TABLE IF EXISTS usuarios;

CREATE TABLE usuarios (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    foto VARCHAR(255),
    nombre VARCHAR(100),
    contrasena VARCHAR(100),
    especialidad VARCHAR(255),
    anios_empresa INT,
    email VARCHAR(255),
    tipo ENUM('admin', 'usuario')
);

CREATE TABLE temas (
    id_tema INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100),
    caracteristica VARCHAR(50),
    imagen VARCHAR(255)
);

CREATE TABLE posts (
    id_post INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT,
    id_tema INT,
    contenido TEXT,
    ruta_media VARCHAR(255),
    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario),
    FOREIGN KEY (id_tema) REFERENCES temas(id_tema)
);

CREATE TABLE respuestas (
    id_respuesta INT PRIMARY KEY AUTO_INCREMENT,
    id_post INT,
    id_usuario INT,
    contenido TEXT,
    ruta_media VARCHAR(255),
    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_post) REFERENCES posts(id_post),
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
);

CREATE TABLE guardado (
    id_guardado INT PRIMARY KEY AUTO_INCREMENT,
    id_post INT,
    id_usuario INT,
    FOREIGN KEY (id_post) REFERENCES posts(id_post),
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
);

CREATE TABLE likeUsuario (
    id_likeUsuario INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT,
    id_respuesta INT,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario),
    FOREIGN KEY (id_respuesta) REFERENCES respuestas(id_respuesta),
    UNIQUE (id_usuario, id_respuesta) -- Evita que el mismo usuario dé like más de una vez a la misma respuesta
);

ALTER TABLE guardado AUTO_INCREMENT = 1;
ALTER TABLE respuestas AUTO_INCREMENT = 1;
ALTER TABLE posts AUTO_INCREMENT = 1;
ALTER TABLE temas AUTO_INCREMENT = 1;
ALTER TABLE usuarios AUTO_INCREMENT = 1;
ALTER TABLE likeUsuario AUTO_INCREMENT = 1;
